def AR_KS(data_XY, prob):
    data_XY = pd.DataFrame(data_XY['budo'])
    data_XY['pred'] = prob
    AR = metrics.roc_auc_score(data_XY['budo'], data_XY['pred']) * 2 * 100 - 100
    good = data_XY.loc[data_XY['budo'] == 0]
    bad = data_XY.loc[data_XY['budo'] == 1]
    KS_tmp = stats.ks_2samp(good['pred'], bad['pred'])
    KS = round(KS_tmp[0], 4) * 100
    prob_attached = data_XY

    return [AR, KS]

def custom_dummy(except_cols, data) : 
    #  
    data = data.astype({'ID' : str, 'sp_month' : str})
    #  
    data_temp = pd.get_dummies(data.drop(except_cols, axis = 1), drop_first=True, dtype='int64')
    #  ID sp_month 
    data_dummies = pd.concat([data[except_cols], data_temp], axis = 1)
    return data_dummies

def data_split(train_list, valid_list,test_list, input_data) :
    # ***_set =   / ***_data = X variables / ***_label = Y variable
    import sys
    mod = sys.modules[__name__]
    
    # <train_data>
    # train_list    train 
    setattr(mod, 'train_set', input_data[input_data['sp_month'].str[:4].isin(train_list)])

    for i in valid_list :

def model_builder(hp, input_shape = train_data.shape[1:]):
    
    # 
    model = keras.Sequential()

    # -    .
    model.add(keras.layers.InputLayer(input_shape = input_shape)) 

    # -     
    #    
    for i in range(hp.Int('num_layers', 2, 20)):
        model.add(layers.Dense(units=hp.Int('units_' + str(i),
                                            min_value=32,
                                            max_value=512,
                                            step=32),
                               activation='relu'
                              ,kernel_initializer = 'he_normal'))
        
    #
    model.add(keras.layers.Dense(1, activation  = 'sigmoid'))

    #(learning_rate ) 
    #Choose an optimal value from 0.01, 0.001, or 0.0001
    hp_learning_rate = hp.Choice('learning_rate', values = [1e-2, 1e-3, 1e-4]) 

    #Model compile
    model.compile(optimizer = keras.optimizers.Adam(learning_rate = hp_learning_rate),
                loss = 'binary_crossentropy', 
                metrics = METRICS)

    return model
